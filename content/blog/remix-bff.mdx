import type { LoaderFunctionArgs } from "@remix-run/node"

export async function loader({
  request,
}: LoaderFunctionArgs) {

  const events = await fetch(
    process.env.API_URL + "/events",
    {
      headers: {
        Authorization: `Bearer ${user.token}`,
      },
    },
  ).then((res) => res.json())
}
